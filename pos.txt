<script>
// === CONFIGURACIÓN ===
const API_URL = 'https://script.google.com/macros/s/AKfycbzNY5A4Rls24akHueBkWLU7VFZ6DcFIkCJETe4Rdx13UUU47BYKF9yQ07fKctBXQUoF/exec';
let carrito = JSON.parse(localStorage.getItem('lh_cart') || '[]');
let pendingSales = JSON.parse(localStorage.getItem('lh_pending') || '[]');
let productos = [];
let clientes = JSON.parse(localStorage.getItem('lh_clients') || '[]');
let ventaActual = null;

// === INIT ===
async function init(){
  renderCart();
  renderPending();
  await loadProductos();
  await loadClientes();
  if(navigator.onLine) syncPending();
  window.addEventListener('online', syncPending);
  
  // Service Worker & PWA
  if('serviceWorker' in navigator){
    const sw = `self.addEventListener('install',e=>self.skipWaiting());self.addEventListener('fetch',e=>e.respondWith(fetch(e.request).catch(()=>new Response('Offline'))));`;
    const blob = new Blob([sw],{type:'application/javascript'});
    navigator.serviceWorker.register(URL.createObjectURL(blob));
  }
}

// === API ===
async function callAPI(accion, params={}){
  const url = new URL(API_URL);
  url.searchParams.set('accion', accion);
  Object.entries(params).forEach(([k,v])=>url.searchParams.set(k, v));
  const res = await fetch(url.toString());
  if(!res.ok) throw new Error('Error de red');
  const data = await res.json();
  if(!data.success) throw new Error(data.error);
  return data;
}

async function loadProductos(){
  try{
    const res = await callAPI('getProductos');
    productos = res.productos || [];
    renderProductos();
    updateProductSelect();
  }catch(e){ console.warn('Productos offline'); }
}

async function loadClientes(){
  try{
    const res = await callAPI('getClientes');
    clientes = res.clientes || clientes;
    localStorage.setItem('lh_clients', JSON.stringify(clientes));
    renderClientes();
    updateClientSelect();
  }catch(e){ updateClientSelect(); }
}

// === POS ===
function updateProductSelect(){
  const sel = document.getElementById('selProducto');
  sel.innerHTML = '<option value="">Seleccionar producto...</option>' + 
    productos.map(p=>`<option value="${p.nombre}" data-precio="${p.precioBotella}" data-stock="${p.stock}">${p.nombre} ($${p.precioBotella}) [${p.stock}]</option>`).join('');
  sel.onchange = ()=>{
    const opt = sel.options[sel.selectedIndex];
    document.getElementById('txtPrecio').value = opt.dataset.precio || '';
  };
}

function updateClientSelect(){
  const sel = document.getElementById('selCliente');
  sel.innerHTML = '<option value="Mostrador">🏪 Mostrador (sin cliente)</option>' +
    clientes.map(c=>`<option value="${c.negocio||''} - ${c.nombre} ${c.apellido}">${c.negocio||''} - ${c.nombre} ${c.apellido}</option>`).join('');
}

function addToCart(){
  const prod = document.getElementById('selProducto').value;
  if(!prod) return toast('⚠️ Selecciona un producto');
  const cant = parseInt(document.getElementById('txtCant').value)||1;
  const precio = parseInt(document.getElementById('txtPrecio').value)||0;
  const item = carrito.find(i=>i.nombre===prod);
  if(item) item.cantidad += cant;
  else carrito.push({nombre:prod, precio, cantidad:cant});
  saveCart(); renderCart();
  document.getElementById('txtCant').value = 1;
  document.getElementById('selProducto').value = '';
  document.getElementById('txtPrecio').value = '';
  toast('✅ Agregado');
}

function renderCart(){
  const view = document.getElementById('cartView');
  const container = document.getElementById('cartItems');
  if(carrito.length===0){view.style.display='none';return;}
  view.style.display='block';
  let total=0, html='';
  carrito.forEach((c,i)=>{
    const sub = c.precio*c.cantidad; total+=sub;
    html+=`<div class="item-row"><div><span class="name">${c.nombre}</span><br><span class="detail">x${c.cantidad} × $${c.precio}</span></div><div><span class="price">$${sub}</span><span class="remove" onclick="removeFromCart(${i})">✕</span></div></div>`;
  });
  container.innerHTML=html;
  document.getElementById('cartTotal').textContent='$'+total;
}

function removeFromCart(i){carrito.splice(i,1);saveCart();renderCart();}
function saveCart(){localStorage.setItem('lh_cart',JSON.stringify(carrito));}

function checkout(){
  if(carrito.length===0) return toast('⚠️ Carrito vacío');
  const pago = document.getElementById('selPago').value;
  if(!pago) return toast('⚠️ Selecciona forma de pago');
  const cliente = document.getElementById('selCliente').value || 'Mostrador';
  const total = carrito.reduce((s,c)=>s+c.precio*c.cantidad,0);
  
  ventaActual = {fecha:new Date().toLocaleString('es-AR'), cliente, carrito:[...carrito], total, pago};
  guardarVentaOffline(ventaActual);
  renderResumen(ventaActual);
  nav('resumen');
  carrito=[]; saveCart(); renderCart();
  document.getElementById('selPago').value='';
  toast('✅ Venta registrada');
}

function guardarVentaOffline(venta){
  pendingSales.push({...venta, status:'pending', timestamp:Date.now()});
  localStorage.setItem('lh_pending', JSON.stringify(pendingSales));
  syncPending();
}

async function syncPending(){
  if(!navigator.onLine || pendingSales.length===0) return;
  toast('🔄 Sincronizando ventas offline...');
  const toSync = [...pendingSales];
  pendingSales = [];
  localStorage.setItem('lh_pending', JSON.stringify(pendingSales));
  
  for(const v of toSync){
    try{
      const params = {
        productos: v.carrito.map(c=>c.nombre).join('|'),
        precios: v.carrito.map(c=>c.precio).join('|'),
        cantidades: v.carrito.map(c=>c.cantidad).join('|'),
        cliente: v.cliente,
        pago: v.pago
      };
      await callAPI('guardarVenta', params);
      
      // Actualizar stocks
      await callAPI('actualizarStocks', {
        nombres: params.productos,
        cantidades: params.cantidades
      });
    }catch(e){
      pendingSales.push({...v, status:'error'});
    }
  }
  localStorage.setItem('lh_pending', JSON.stringify(pendingSales));
  toast('✅ Sincronización completa');
}

function showClientForm(){document.getElementById('clientForm').style.display='block';}
function hideClientForm(){document.getElementById('clientForm').style.display='none';}
async function saveClient(){
  const nom=document.getElementById('cliNom').value.trim();
  const tel=document.getElementById('cliTel').value.trim();
  if(!nom||!tel) return toast('⚠️ Nombre y teléfono son obligatorios');
  const cli = {
    id:'C'+Date.now(), nombre:nom, apellido:document.getElementById('cliApe').value.trim(),
    negocio:document.getElementById('cliNeg').value.trim(), tipo:document.getElementById('cliTipo').value,
    telefono:tel, direccion:document.getElementById('cliDir').value.trim(),
    cuit:document.getElementById('cliCUIT').value.trim(), notas:document.getElementById('cliNotas').value.trim()
  };
  try{ await callAPI('guardarCliente', cli); }catch(e){}
  clientes.push(cli); localStorage.setItem('lh_clients',JSON.stringify(clientes));
  renderClientes(); updateClientSelect(); hideClientForm(); toast('✅ Cliente guardado');
  ['cliNom','cliApe','cliNeg','cliTel','cliDir','cliCUIT','cliNotas'].forEach(id=>document.getElementById(id).value='');
}

      <div class="detail">${c.negocio||'Sin negocio'} | 📞 ${c.telefono}</div>
      <div class="detail">📍 ${c.direccion||'Sin dirección'} | 🆔 ${c.cuit||'Sin CUIT'}</div>
      ${c.notas?`<div class="detail">📝 ${c.notas}</div>`:''}
    </div>`).join('');
}

async function saveProduct(){
  const nom=document.getElementById('prodNom').value.trim();
  const pre=parseInt(document.getElementById('prodPre').value);
  const sto=parseInt(document.getElementById('prodSto').value);
  if(!nom||isNaN(pre)||isNaN(sto)) return toast('⚠️ Completá nombre, precio y stock');
  try{
    await callAPI('guardarProducto', {nombre:nom, estilo:document.getElementById('prodEst').value.trim(), precioBotella:pre, precioPack:pre*6, stock:sto});
    await loadProductos(); hideProdForm(); toast('✅ Producto guardado');
  }catch(e){toast('❌ Error al guardar');}
  ['prodNom','prodEst','prodPre','prodSto'].forEach(id=>document.getElementById(id).value='');
}
